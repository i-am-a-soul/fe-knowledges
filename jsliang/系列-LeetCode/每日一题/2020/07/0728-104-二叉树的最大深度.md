## 三 解题思路



* 递归

```js
/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
/**
 * 思考，方法两种：
 * 1. 递归
 * 2. 遍历
 */
const maxDepth = (root, depth = 1) => {
  if (!root) {
    return 0;
  }
  let tempDepth = depth;
  if (root.left) {
    tempDepth = Math.max(tempDepth, maxDepth(root.left, depth + 1));
  }
  if (root.right) {
    tempDepth = Math.max(tempDepth, maxDepth(root.right, depth + 1));
  }
  return tempDepth;
};

/*
    3
   / \
  9  20
    /  \
   15   7
*/
const root = {
  val: 3,
  left: { val: 9, left: null, right: null },
  right: {
    val: 20,
    left: { val: 15, left: null, right: null },
    right: { val: 7, left: null, right: null },
  },
};
console.log(maxDepth(root)); // 3
```

* 迭代

```js
/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
/**
 * 思考，方法两种：
 * 1. 递归
 * 2. 遍历
 */
const maxDepth = (root) => {
  if (!root) {
    return 0;
  }
  let depth = 0;
  let cursor = [root];
  while (cursor.length) {
    const tempRoot = [];
    for (let i = 0; i < cursor.length; i++) {
      const nowRoot = cursor[i];
      if (nowRoot.left) {
        tempRoot.push(nowRoot.left);
      }
      if (nowRoot.right) {
        tempRoot.push(nowRoot.right);
      }
    }
    depth++;
    cursor = tempRoot;
  }
  return depth;
};
```
