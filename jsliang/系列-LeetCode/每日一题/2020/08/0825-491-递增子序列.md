```
给定一个整型数组, 
你的任务是找到所有该数组的递增子序列，
递增子序列的长度至少是2。

示例:

输入: [4, 6, 7, 7]
输出: [
  [4, 6], [4, 7], [4, 6, 7],

]
```

## 三 解题思路



递归：

```js
/**
 * @param {number[]} nums
 * @return {number[][]}
 */
const findSubsequences = (nums) => {
  // 设置结果
  const result = [];

  // 通过 Set 存储已经出现的子序列
  const set = new Set();

  // 递归探索
  const recursion = (start, path) => {
    if (path.length >= 2) {
      const str = path.join('|'); // 路径转成字符串
      // 如果 Set 中不包含这个字符串
      if (!set.has(str)) {
        // 那么 Set 添加该字符串
        set.add(str);
        // 通过 slice 复制当前数组，不能直接添加进去
        // 因为 path 是传递地址的数组，最后会被清空或者同化
        result.push(path.slice());
      }
    }
    for (let i = start; i < nums.length; i++) {
      const prev = path[path.length - 1];
      const current = nums[i];
      if (!path.length || prev <= current) {
        path.push(current); // 选择当前的数字
        recursion(i + 1, path); // 继续往下递归
        path.pop(); // 撤销当前数字，选择别的数字
      }
    }
  }
  recursion(0, []);

  // 返回结果
  return result;
};

console.log(findSubsequences([4, 6, 7, 7]));
/*
[
  [ 4, 6 ],
  [ 4, 6, 7 ],
  [ 4, 6, 7, 7 ],
  [ 4, 7 ],
  [ 4, 7, 7 ],
  [ 6, 7 ],
  [ 6, 7, 7 ],
  [ 7, 7 ],
]
*/
```
