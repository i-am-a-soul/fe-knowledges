## 三 解题思路



栈：

```js
/**
 * @param {string} s
 * @return {boolean}
 */
const isValid = (s) => {
  // 1. 设置栈 
  const stack = [];

  // 2. 遍历入栈和出栈
  for (let i = 0; i < s.length; i++) {
    const tail = stack[stack.length - 1]; // 获取栈顶    
    // 如果栈顶元素和当前元素构成一对括号，则消除，否则添加
    if (
      (s[i] === ')' && tail === '(')
      || (s[i] === ']' && tail === '[')
      || (s[i] === '}' && tail === '{')
    ) {
      stack.pop();
    } else {
      stack.push(s[i]);
    }
  }

  // 3. 判断栈是否还有长度，如果没有了，说明符合预期
  return !stack.length;
};

console.log(isValid('[]{}()'))
```
