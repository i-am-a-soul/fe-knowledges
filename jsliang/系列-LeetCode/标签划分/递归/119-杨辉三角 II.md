## 三 解题思路



递归：

```js
/**
 * @param {number} rowIndex
 * @return {number[]}
 */
const getRow = (rowIndex) => {
  if (rowIndex === 0) {
    return [1];
  } else if (rowIndex === 1) {
    return [1, 1];
  }
  const prevRow = getRow(rowIndex - 1);
  const nowRow = Array.from(Array(rowIndex + 1), () => 1);
  for (let i = 1; i < nowRow.length - 1; i++) {
    nowRow[i] = prevRow[i] + prevRow[i - 1];
  }
  return nowRow;
};

console.log(getRow(3));
```
