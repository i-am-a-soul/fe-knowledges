## 三 解题思路



使用 `Set`：

```js
/**
 * @param {number[]} nums
 * @return {boolean}
 */
const containsDuplicate = (nums) => {
  // 1. 因为需要判断唯一，所以设置 set
  const set = new Set();

  // 2. 顺序遍历，如果没有就添加，如果有就返回 true
  for (let i = 0; i < nums.length; i++) {
    if (!set.has(nums[i])) {
      set.add(nums[i]);
    } else {
      return true;
    }
  }

  // 3. 否则返回 false
  return false;
};

console.log(containsDuplicate([1, 2, 3, 1])); // true
```
